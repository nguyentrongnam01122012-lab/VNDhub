-- [1. CẤU HÌNH TỌA ĐỘ]
local TDChotCongAn = Vector3.new(734, 20, -812)
local TDBoss = Vector3.new(-2823, 274, -1896)

-- [2. CÁC HÀM HỖ TRỢ]
local function isAreaOccupied(targetPos)
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            if (p.Character.HumanoidRootPart.Position - targetPos).Magnitude <= 7 then return true end
        end
    end
    for _, v in ipairs(workspace:GetDescendants()) do
        if v:IsA("Seat") or v:IsA("VehicleSeat") then
            if (v.Position - targetPos).Magnitude <= 7 and v.Occupant and v.Occupant.Parent ~= player.Character then return true end
        end
    end
    return false
end

local function TeleportTo(targetPos)
    local char = player.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        for _, part in pairs(char:GetChildren()) do if part:IsA("BasePart") then part.CanCollide = false end end
        char.HumanoidRootPart.CFrame = CFrame.new(targetPos)
        task.wait(0.5)
        for _, part in pairs(char:GetChildren()) do if part:IsA("BasePart") then part.CanCollide = true end end
    end
end

-- [3. ANTI-AFK]
player.Idled:Connect(function()
    VirtualUser:CaptureController()
    VirtualUser:ClickButton2(Vector2.new())
end)

-- [4. MAIN LOOP - AUTO SIT & TELEPORT]
task.spawn(function()
    while task.wait(5) do
        local char = player.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        
        if hum and not hum.Sit then
            -- Tự động Tele đến Chốt Công An nếu khu vực trống
            if not isAreaOccupied(TDChotCongAn) then
                TeleportTo(TDChotCongAn)
            end
        end
    end
end)

-- [5. AUTO RECONNECT]
task.spawn(function()
    while task.wait(10) do
        if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then
            TeleportService:Teleport(game.PlaceId, player)
        end
    end
end)
