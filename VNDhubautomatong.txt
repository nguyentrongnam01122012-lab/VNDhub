-- [2. AUTO SIT - ĐÃ FIX DỊCH CHUYỂN VÀ RUNG]
local targetPositions = {
    Vector3.new(567, 19, -533), 
    Vector3.new(626, 19, -511), 
    Vector3.new(629, 19, -419)
}

task.spawn(function()
    while task.wait(2) do
        local char = plr.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        
        if hrp and hum and hum.Health > 0 and not hum.Sit then
            for _, pos in pairs(targetPositions) do
                local foundSeat = nil
                
                -- Tìm ghế gần tọa độ
                for _, obj in pairs(workspace:GetDescendants()) do
                    if obj:IsA("Seat") and (obj.Position - pos).Magnitude < 10 then
                        if not obj.Occupant then
                            foundSeat = obj
                            break
                        end
                    end
                end
                
                -- Nếu thấy ghế trống, di chuyển tới
                if foundSeat then
                    -- Kiểm tra khoảng cách: nếu đã sát ghế (< 4 stud) thì không cần Teleport nữa
                    if (hrp.Position - foundSeat.Position).Magnitude > 4 then
                        hrp.CFrame = foundSeat.CFrame
                        task.wait(0.5) -- Đợi nhân vật chạm ghế
                    end
                    
                    foundSeat:Sit(hum)
                    task.wait(2) -- Nghỉ 2 giây để tránh rung sau khi đã ngồi
                    break
                end
            end
        end
    end
end)
