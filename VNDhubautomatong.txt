--[[ 
  💰 VND💵 - Pro Edition (Full Auto + Auto Click khi không ở đúng vị trí)
]]

local plr = game:GetService("Players").LocalPlayer
local cam = workspace.CurrentCamera
local vu = game:GetService("VirtualUser")
local CoreGui = game:GetService("CoreGui")
local TeleportService = game:GetService("TeleportService")

-- [CẤU HÌNH VỊ TRÍ]
local targetPositions = {
    Vector3.new(567, 19, -533), 
    Vector3.new(626, 19, -511), 
    Vector3.new(629, 19, -419)
}

-- [1. ANTI-AFK ĐẲNG CẤP]
local function startAntiAFK()
    plr.Idled:Connect(function()
        vu:CaptureController()
        vu:ClickButton2(Vector2.new())
    end)
    task.spawn(function()
        while task.wait(60) do
            vu:CaptureController()
        end
    end)
end
startAntiAFK()

-- [2. AUTO SIT + 3. EMERGENCY INTERACT (Ấn 5 lần)]
task.spawn(function()
    while task.wait(2) do
        local char = plr.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        
        if hrp and hum then
            local isNearTarget = false
            for _, pos in pairs(targetPositions) do
                if (hrp.Position - pos).Magnitude < 15 then
                    isNearTarget = true
                    break
                end
            end

            -- Nếu không ở gần tọa độ mục tiêu: Thực hiện 5 lần click
            if not isNearTarget then
                for i = 1, 5 do
                    vu:Button1Down(Vector2.new(500, 500))
                    task.wait(0.1)
                    vu:Button1Up(Vector2.new(500, 500))
                end
            elseif not hum.Sit then
                -- Nếu ở gần mục tiêu nhưng chưa ngồi: Thực hiện ngồi
                for _, pos in pairs(targetPositions) do
                    for _, obj in pairs(workspace:GetDescendants()) do
                        if obj:IsA("Seat") and (obj.Position - pos).Magnitude < 15 then
                            if not obj.Occupant then
                                hrp.CanCollide = false
                                hrp.CFrame = obj.CFrame
                                task.wait(0.1)
                                obj:Sit(hum)
                                task.wait(0.5)
                                hrp.CanCollide = true
                                break
                            end
                        end
                    end
                end
            end
        end
    end
end)

-- [4. AUTO RECONNECT]
task.spawn(function()
    while task.wait(5) do
        local errorGui = CoreGui:FindFirstChild("RobloxPromptGui")
        if errorGui and errorGui:FindFirstChild("ErrorMessage", true) then
            TeleportService:Teleport(game.PlaceId, plr)
        end
    end
end)

-- [5. UI VND💵]
local gui = Instance.new("ScreenGui", plr:WaitForChild("PlayerGui"))
local f = Instance.new("Frame", gui)
f.Size = UDim2.new(0, 140, 0, 40)
f.Position = UDim2.new(0, 20, 0, 20)
f.BackgroundColor3 = Color3.fromRGB(15, 15, 20)
f.Active = true 
f.Draggable = true
Instance.new("UICorner", f).CornerRadius = UDim.new(0, 8)
Instance.new("UIStroke", f).Color = Color3.fromRGB(255, 215, 0)

local title = Instance.new("TextLabel", f)
title.Size = UDim2.new(1, 0, 1, 0)
title.Text = "VND💵 ACTIVE"
title.Font = Enum.Font.Code
title.TextColor3 = Color3.fromRGB(255, 215, 0)
title.BackgroundTransparency = 1
