  Tính năng: Anti-AFK, Auto Sit (Fix lỗi), Auto Click (Emergency), Auto Reconnect

local Players = game:GetService("Players")
local player = Players.LocalPlayer
local VirtualUser = game:GetService("VirtualUser")
local CoreGui = game:GetService("CoreGui")
local TeleportService = game:GetService("TeleportService")
local workspace = game:GetService("Workspace")

-- [1. CẤU HÌNH TỌA ĐỘ CỦA BẠN]
local targetPositions = {
    Vector3.new(573, 21, -357),
    Vector3.new(563, 20, -386),
    Vector3.new(553, 20, -413)
}

-- [2. CÁC HÀM HỖ TRỢ]
local function isAreaOccupied(targetPos)
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            if (p.Character.HumanoidRootPart.Position - targetPos).Magnitude <= 7 then return true end
        end
    end
    for _, v in ipairs(workspace:GetDescendants()) do
        if (v:IsA("Seat") or v:IsA("VehicleSeat")) then
            if (v.Position - targetPos).Magnitude <= 7 and v.Occupant and v.Occupant.Parent ~= player.Character then 
                return true 
            end
        end
    end
    return false
end

-- [3. AUTO SIT + EMERGENCY CLICK 5 LẦN]
task.spawn(function()
    local hasClicked = false
    while task.wait(2) do
        local char = player.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        
        if hrp and hum then
            local isNear = false
            for _, pos in pairs(targetPositions) do
                if (hrp.Position - pos).Magnitude < 15 then isNear = true break end
            end

            -- Nếu ở xa khu vực cày: Click 5 lần để cứu kẹt
            if not isNear then
                if not hasClicked then
                    for i = 1, 5 do
                        VirtualUser:Button1Down(Vector2.new(500, 500))
                        task.wait(0.1)
                        VirtualUser:Button1Up(Vector2.new(500, 500))
                    end
                    hasClicked = true
                end
            else
                hasClicked = false -- Reset trạng thái khi đã về đúng khu vực
                
                -- Nếu ở gần mà chưa ngồi: Ngồi ghế ngay lập tức
                if not hum.Sit then
                    for _, pos in pairs(targetPositions) do
                        if not isAreaOccupied(pos) then
                            hrp.CanCollide = false
                            hrp.Anchored = true
                            hrp.CFrame = CFrame.new(pos)
                            task.wait(0.3)
                            hrp.Anchored = false
                            hrp.CanCollide = true
                            break
                        end
                    end
                end
            end
        end
    end
end)

-- [4. ANTI-AFK]
player.Idled:Connect(function()
    VirtualUser:CaptureController()
    VirtualUser:ClickButton2(Vector2.new())
end)

-- [5. AUTO RECONNECT]
task.spawn(function()
    while task.wait(5) do
        local errorGui = CoreGui:FindFirstChild("RobloxPromptGui")
        if errorGui and errorGui:FindFirstChild("ErrorMessage", true) then
            TeleportService:Teleport(game.PlaceId, player)
        end
    end
end)
