-- CẤU HÌNH LOGIC AUTO FARM (Đã thêm biến bật/tắt)
------------------------------------------------
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local VirtualUser = game:GetService("VirtualUser")

local player = Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()
local hrp = char:WaitForChild("HumanoidRootPart")

local currentBoss = nil
local attackInterval = 0.1 
local lastAttack = 0
local moveSpeed = 8        

_G.AutoFarmLevel = false -- Biến trạng thái bật/tắt hệ thống

local function isAliveModel(model)
    if model and model:FindFirstChild("Humanoid") and model.Humanoid.Health > 0 then
        return true
    end
    return false
end

local function getNearestBoss(hrpPos)
    local closestBoss = nil
    local maxDistance = math.huge
    local bossFolder = workspace:FindFirstChild("Bosses") or workspace
    
    for _, obj in ipairs(bossFolder:GetChildren()) do
        if obj:IsA("Model") and isAliveModel(obj) and obj ~= char then
            local bossHrp = obj:FindFirstChild("HumanoidRootPart")
            if bossHrp then
                local distance = (hrpPos - bossHrp.Position).Magnitude
                if distance < maxDistance then
                    maxDistance = distance
                    closestBoss = obj
                end
            end
        end
    end
    return closestBoss
end

-- Vòng lặp chính kiểm tra điều kiện công tắc
RunService.Heartbeat:Connect(function(dt)
    -- Nếu công tắc TẮT thì dừng toàn bộ xử lý bên dưới
    if not _G.AutoFarmLevel then 
        currentBoss = nil
        return 
    end

    if not char or not char:FindFirstChild("HumanoidRootPart") then 
        char = player.Character
        if char then hrp = char:FindFirstChild("HumanoidRootPart") end
        return 
    end

    local nearest = getNearestBoss(hrp.Position)
    
    if nearest and nearest ~= currentBoss and isAliveModel(nearest) then
        currentBoss = nearest
    end

    if currentBoss and isAliveModel(currentBoss) then
        local bossHrp = currentBoss:FindFirstChild("HumanoidRootPart")
        if bossHrp then
            hrp.CFrame = hrp.CFrame:Lerp(CFrame.new(bossHrp.Position + Vector3.new(0, 3, 0), bossHrp.Position), dt * moveSpeed)
            
            lastAttack = lastAttack + dt
            if lastAttack >= attackInterval then
                lastAttack = 0
                VirtualUser:ClickButton1(Vector2.new())
            end
        end
    else
        currentBoss = nil
    end
end)
