local function isAliveModel(model)
    if model and model:FindFirstChild("Humanoid") and model.Humanoid.Health > 0 then
        return true
    end
    return false
end

local function getNearestMonster()
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local hrpPos = char.HumanoidRootPart.Position
    local closestMonster = nil
    local maxDistance = math.huge
    local targetFolder = workspace:FindFirstChild("Bosses") or workspace:FindFirstChild("Monsters") or workspace
    
    for _, obj in ipairs(targetFolder:GetChildren()) do
        if obj:IsA("Model") and isAliveModel(obj) and obj ~= char then
            if not Players:GetPlayerFromCharacter(obj) then 
                local mHrp = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Torso")
                if mHrp then
                    local distance = (hrpPos - mHrp.Position).Magnitude
                    if distance < maxDistance then
                        maxDistance = distance
                        closestMonster = obj
                    end
                end
            end
        end
    end
    return closestMonster
end

-- Tìm kiếm người chơi khác gần nhất để Aim PvP
local function getNearestPlayer()
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local hrpPos = char.HumanoidRootPart.Position
    local closestPlayer = nil
    local maxDistance = math.huge
    
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") and isAliveModel(p.Character) then
            local distance = (hrpPos - p.Character.HumanoidRootPart.Position).Magnitude
            if distance < maxDistance then
                maxDistance = distance
                closestPlayer = p.Character
            end
        end
    end
    return closestPlayer
end

local function isAreaOccupied(targetPos)
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            if (p.Character.HumanoidRootPart.Position - targetPos).Magnitude <= 7 then
                return true
            end
        end
    end
    for _, v in ipairs(workspace:GetDescendants()) do
        if v:IsA("Seat") or v:IsA("VehicleSeat") then
            if (v.Position - targetPos).Magnitude <= 7 and v.Occupant and v.Occupant.Parent ~= player.Character then 
                return true 
            end
        end
    end
    return false
end

local function interactWithSeatPrompt(targetPos)
    for _, prompt in ipairs(workspace:GetDescendants()) do
        if prompt:IsA("ProximityPrompt") and (prompt.Parent:IsA("BasePart") or prompt.Parent:IsA("Model")) then
            if (prompt.Parent.Position - targetPos).Magnitude <= 6 then
                fireproximityprompt(prompt)
                break
            end
        end
    end
end

local function activateFPSBoost()
    settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
    game:GetService("Lighting").GlobalShadows = false
    game:GetService("Lighting").Brightness = 0
    game:GetService("Lighting").FogEnd = 1e9
    for _,v in ipairs(workspace:GetDescendants()) do
        if v:IsA("Decal") or v:IsA("Texture") or v:IsA("Sound") then pcall(function() v:Destroy() end) end
        if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") then v.Enabled = false end
        if v:IsA("BasePart") then
            v.Material = Enum.Material.SmoothPlastic
            v.Reflectance = 0
            v.CastShadow = false
        end
    end
end

local function deactivateFPSBoost()
    game:GetService("Lighting").GlobalShadows = true
    game:GetService("Lighting").Brightness = 2
    game:GetService("Lighting").FogEnd = 1000
end

-- ❤️ LUỒNG CORE HEALTH ESP: Tạo và cập nhật thanh máu trên đầu người chơi khác
local function createHealthESP(p)
    if p == player then return end
    
    local function setupESP(char)
        local head = char:WaitForChild("Head", 5)
        local humanoid = char:WaitForChild("Humanoid", 5)
        if not head or not humanoid then return end
        
        if head:FindFirstChild("PlayerHealthESP") then
            head.PlayerHealthESP:Destroy()
        end
        
        local bGui = Instance.new("BillboardGui")
        bGui.Name = "PlayerHealthESP"
        bGui.Size = UDim2.new(0, 130, 0, 30)
        bGui.AlwaysOnTop = true
        bGui.ExtentsOffset = Vector3.new(0, 2.5, 0)
        bGui.Parent = head
        
        local bgFrame = Instance.new("Frame")
        bgFrame.Size = UDim2.new(1, 0, 0, 8)
        bgFrame.Position = UDim2.new(0, 0, 0, 18)
        bgFrame.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
        bgFrame.BorderSizePixel = 1
        bgFrame.Parent = bGui
        
        local fillFrame = Instance.new("Frame")
        fillFrame.Size = UDim2.new(1, 0, 1, 0)
        fillFrame.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
        fillFrame.BorderSizePixel = 0
        fillFrame.Parent = bgFrame
        
        local textLabel = Instance.new("TextLabel")
        textLabel.Size = UDim2.new(1, 0, 0, 14)
        textLabel.BackgroundTransparency = 1
        textLabel.TextSize = 12
        textLabel.Font = Enum.Font.GothamBold
        textLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
        textLabel.TextStrokeTransparency = 0
        textLabel.Parent = bGui
        
        local connection
        connection = RunService.Heartbeat:Connect(function()
            if not _G.PvpBậtHiệnMáu or not char.Parent or not humanoid or humanoid.Health <= 0 then
                bGui:Destroy()
                connection:Disconnect()
                return
            end
            
            local hpPercentage = math.clamp(humanoid.Health / humanoid.MaxHealth, 0, 1)
            fillFrame.Size = UDim2.new(hpPercentage, 0, 1, 0)
            
            fillFrame.BackgroundColor3 = Color3.fromHSV(hpPercentage * 0.33, 1, 1)
            textLabel.Text = p.Name .. " [" .. math.floor(humanoid.Health) .. "/" .. math.floor(humanoid.MaxHealth) .. "]"
        end)
    end
    
    if p.Character then task.spawn(setupESP, p.Character) end
    p.CharacterAdded:Connect(function(char)
        if _G.PvpBậtHiệnMáu then task.spawn(setupESP, char) end
    end)
end

for _, p in ipairs(Players:GetPlayers()) do createHealthESP(p) end
Players.PlayerAdded:Connect(createHealthESP)

-- 🎥 Vòng lặp 8: Quản lý góc nhìn tự do + Xoay Tròn + Aim Tĩnh PvP độc lập
RunService.RenderStepped:Connect(function(deltaTime)
    pcall(function()
        local char = player.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")

        if hrp then
            if _G.PvpSpinActive then
                hrp.CFrame *= CFrame.Angles(0, math.rad(spinSpeed * deltaTime), 0)
            elseif _G.PvpAutoAimTĩnh then
                local pTarget = getNearestPlayer()
                if pTarget and pTarget:FindFirstChild("HumanoidRootPart") then
                    local targetPos = pTarget.HumanoidRootPart.Position
                    local lookAtFlat = Vector3.new(targetPos.X, hrp.Position.Y, targetPos.Z)
                    hrp.CFrame = CFrame.lookAt(hrp.Position, lookAtFlat)
                end
            end
        end

        if _G.LockCameraEnabled then
            camera.CameraType = Enum.CameraType.Scriptable
        elseif _G.FreecamEnabled then
            camera.CameraType = Enum.CameraType.Scriptable
            
            local lookVector = camera.CFrame.LookVector
            local rightVector = camera.CFrame.RightVector
            local moveVector = Vector3.new()
            
            if UserInputService:IsKeyDown(Enum.KeyCode.W) or UserInputService:IsKeyDown(Enum.KeyCode.Up) then moveVector = moveVector + lookVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.S) or UserInputService:IsKeyDown(Enum.KeyCode.Down) then moveVector = moveVector - lookVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.A) or UserInputService:IsKeyDown(Enum.KeyCode.Left) then moveVector = moveVector - rightVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.D) or UserInputService:IsKeyDown(Enum.KeyCode.Right) then moveVector = moveVector + rightVector end
            
            if moveVector.Magnitude > 0 then
                freecamPosition = freecamPosition + (moveVector.Unit * flySpeed)
            end
            
            camera.CFrame = CFrame.new(freecamPosition) * camera.CFrame.Rotation
        else
            if camera.CameraType == Enum.CameraType.Scriptable then
                camera.CameraType = Enum.CameraType.Custom
            end
        end
    end)
end)

-- ⚙️ Thêm công tắc Chạy nhanh & viết số tùy biến
Tabs.Main5:AddToggle("WalkSpeedToggle", {
    Title = "⚡ Bật Chạy Nhanh", 
    Default = false, 
    Callback = function(Value)
        _G.BậtChạyNhanh = Value
        if not Value then
            pcall(function() player.Character.Humanoid.WalkSpeed = 16 end) -- Trả lại tốc độ mặc định khi tắt
        end
    end
})

Tabs.Main5:AddInput("WalkSpeedInput", {
    Title = "Nhập tốc độ chạy mong muốn",
    Default = "16",
    Placeholder = "Gõ số vào đây...",
    Numeric = true, -- Chỉ cho gõ số
    Finished = true, -- Chỉ cập nhật khi ấn Enter
    Callback = function(Value)
        _G.TốcĐộChạy = tonumber(Value) or 16
    end
})

-- ⚙️ Thêm công tắc Nhảy cao & viết số tùy biến

    Title = "🦘 Bật Nhảy Cao", 
    Default = false, 
    Callback = function(Value)
        _G.BậtNhảyCao = Value

        end
    end
})
