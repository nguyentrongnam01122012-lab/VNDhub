-- ====== GENERATE KEY ======
local function generateUserKey()
    local period = math.floor(os.time() / 3600 / PERIOD_HOURS)
    local seed   = (period * 9999 + tonumber(userId)) % INT32_MAX
    math.randomseed(seed)
    local s = ""
    for _ = 1, KEY_LENGTH do
        s = s .. tostring(math.random(0, 9))
    end
    return KEY_PREFIX .. s
end

local function isValidKeyFormat(k)
    if type(k) ~= "string" then return false end
    return k:match("^VND_[%w_]+$") ~= nil
end

-- ====== PARSE EXPIRE ======
local function parseExpireAt(val)
    if val == nil then return 0 end
    local n = tonumber(val)
    if n then return n end
    local s = tostring(val)
    local y, m, d = s:match("^(%d%d%d%d)-(%d%d)-(%d%d)")
    if y then
        return os.time({year=tonumber(y),month=tonumber(m),day=tonumber(d),hour=23,min=59,sec=59})
    end
    local d2, m2, y2 = s:match("^(%d%d)/(%d%d)/(%d%d%d%d)")
    if d2 then
        return os.time({year=tonumber(y2),month=tonumber(m2),day=tonumber(d2),hour=23,min=59,sec=59})
    end
    return 0
end

-- ====== FIX: Xử lý Firebase trả về array node ======
local function resolveFirebaseData(data)
    if data.key ~= nil or data.expireAt ~= nil or data.expire ~= nil or data.type ~= nil then
        return data
    end
    local latest = nil
    local latestExp = -1
    for _, v in pairs(data) do
        if type(v) == "table" then
            local exp = parseExpireAt(v.expireAt or v.expire)
            if exp > latestExp then
                latestExp = exp
                latest = v
            end
        end
    end
    return latest
end

-- ====== FIREBASE ======
local function saveKeyToFirebase(key)
    local keyId      = key:gsub("[^%w_]", "_")
    local url        = DB .. "/keys/" .. keyId .. ".json"
    local expireTime = os.time() + KEY_DURATION_SEC

    local data = HttpService:JSONEncode({
        key      = key,
        userId   = userId,
        expireAt = expireTime,
        type     = "normal",
        created  = os.date("%d/%m/%Y %H:%M")
    })

    local ok, err = pcall(function()
        HttpService:RequestAsync({
            Url     = url,
            Method  = "PUT",
            Headers = { ["Content-Type"] = "application/json" },
            Body    = data
        })
    end)
    if not ok then warn("[VND] Firebase lỗi: " .. tostring(err)) end
    return ok
end

local function updateKeyWithUserId(key)
    local keyId = key:gsub("[^%w_]", "_")
    local url   = DB .. "/keys/" .. keyId .. ".json"

    local data = HttpService:JSONEncode({
        userId = userId
    })

    local ok, err = pcall(function()
        HttpService:RequestAsync({
            Url     = url,
            Method  = "PATCH",
            Headers = { ["Content-Type"] = "application/json" },
            Body    = data
        })
    end)
    if not ok then warn("[VND] Firebase update lỗi: " .. tostring(err)) end
    return ok
end

local function isAliveModel(model)
    if model and model:FindFirstChild("Humanoid") and model.Humanoid.Health > 0 then

local function getNearestMonster()
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local hrpPos = char.HumanoidRootPart.Position
    local closestMonster = nil
    local maxDistance = math.huge
    local targetFolder = workspace:FindFirstChild("Bosses") or workspace:FindFirstChild("Monsters") or workspace
    
    for _, obj in ipairs(targetFolder:GetChildren()) do
        if obj:IsA("Model") and isAliveModel(obj) and obj ~= char then
            if not Players:GetPlayerFromCharacter(obj) then 
                local mHrp = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Torso")
                if mHrp then
                    local distance = (hrpPos - mHrp.Position).Magnitude
                    if distance < maxDistance then
                        maxDistance = distance
                        closestMonster = obj
                    end
                end
            end
        end
    end
    return closestMonster
end

local originalMaterials = {}

local function saveOriginalMaterials()
    if next(originalMaterials) == nil then
        for _, v in ipairs(workspace:GetDescendants()) do
            if v:IsA("BasePart") then
                originalMaterials[v] = {
                    Material = v.Material,
                    Reflectance = v.Reflectance,
                    CastShadow = v.CastShadow
                }
            end
        end
    end
end

local function toggleGraphicReduction(percent, enable)
    saveOriginalMaterials()
    local totalParts = 0
    for _ in pairs(originalMaterials) do totalParts = totalParts + 1 end
    if totalParts == 0 then return end

    local targetCount = math.floor(totalParts * (percent / 100))
    local currentCount = 0

    for part, data in pairs(originalMaterials) do
        if part and part.Parent then
            currentCount = currentCount + 1
            if currentCount <= targetCount then
                if enable then
                    part.Material = Enum.Material.SmoothPlastic
                    part.Reflectance = 0
                    part.CastShadow = false
                else
                    part.Material = data.Material
                    part.Reflectance = data.Reflectance
                    part.CastShadow = data.CastShadow
                end
            end
        end
    end
end

local ArenaCenter = Vector3.new(-2790, 233, -1760)

local ArenaRadius = 55

local CameraHeight = 60

local OrbitSpeed = 0.3

local currentAngle = 0

local cameraConnection = nil

local cameraPart = nil

local function getBoss2()

    local gh2 = workspace:FindFirstChild("GiangHo2")

    if not gh2 then return nil end

    local npcs = gh2:FindFirstChild("NPCs")

    if not npcs then return nil end

    return npcs:FindFirstChild("NPC2")

end

local function SetupArenaCamera(enabled)

    if enabled then

        if cameraConnection then

            cameraConnection:Disconnect()

            cameraConnection = nil

        end

        if cameraPart then

            cameraPart:Destroy()

            cameraPart = nil

        end

        

        cameraPart = Instance.new("Part")

        cameraPart.Name = "ArenaCameraAnchor"

        cameraPart.Size = Vector3.new(1, 1, 1)

        cameraPart.Transparency = 1

        cameraPart.CanCollide = false

        cameraPart.Anchored = true

        cameraPart.Parent = workspace

        

        workspace.CurrentCamera.CameraType = Enum.CameraType.Scriptable

        currentAngle = 0

        

        cameraConnection = RunService.RenderStepped:Connect(function(deltaTime)

            if enabled then

                currentAngle = currentAngle + (OrbitSpeed * deltaTime)

                

                local offsetX = math.cos(currentAngle) * ArenaRadius

                local offsetZ = math.sin(currentAngle) * ArenaRadius

                

                local currentCenter = ArenaCenter

                local liveBoss = getBoss2()

                if liveBoss and liveBoss:FindFirstChild("HumanoidRootPart") then

                    currentCenter = liveBoss.HumanoidRootPart.Position

                end

                

                local cameraPos = Vector3.new(

                    currentCenter.X + offsetX,

                    currentCenter.Y + CameraHeight,

                    currentCenter.Z + offsetZ

                )

                

                cameraPart.CFrame = CFrame.new(cameraPos, currentCenter)

                workspace.CurrentCamera.CFrame = cameraPart.CFrame

                workspace.CurrentCamera.CameraSubject = cameraPart

            end

        end)

    else

        if cameraConnection then

            cameraConnection:Disconnect()

            cameraConnection = nil

        end

        if cameraPart then

            cameraPart:Destroy()

            cameraPart = nil

        end

        workspace.CurrentCamera.CameraType = Enum.CameraType.Custom

        if Player.Character then

            workspace.CurrentCamera.CameraSubject = Player.Character:FindFirstChildOfClass("Humanoid")

        end

    end

end

local function clearAllHitboxes()
    for name, adornment in pairs(activeHitboxes) do

    for name, gui in pairs(activeHealthGuis) do

local function getClosestPlayer()
    local closestPlayer = nil
    local shortestDistance = math.huge
    local myRoot = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    if myRoot then
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health > 0 then
                local distance = (myRoot.Position - p.Character.HumanoidRootPart.Position).Magnitude
                if distance < shortestDistance then
                    shortestDistance = distance
                    closestPlayer = p
                end
            end
        end
    end
    return closestPlayer
end

RunService.RenderStepped:Connect(function()
    pcall(function()
        local char = player.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChildOfClass("Humanoid")

        -- 1. Siêu Tốc Độ (CFrame Method - An toàn, không bị kick)
        if _G.PvpSpeedActive and hrp and hum and hum.MoveDirection.Magnitude > 0 then
            hrp.CFrame = hrp.CFrame + (hum.MoveDirection * 3)
        end

        -- 2. Spin Logic
        if _G.PvpSpinActive and hrp then
            if not hrp:FindFirstChild("VND_SpinVelocity") then
                local spin = Instance.new("BodyAngularVelocity")
                spin.Name = "VND_SpinVelocity"
                spin.MaxTorque = Vector3.new(0, math.huge, 0)
                spin.AngularVelocity = Vector3.new(0, 40, 0)
                spin.Parent = hrp
            end
        else
            if hrp and hrp:FindFirstChild("VND_SpinVelocity") then hrp["VND_SpinVelocity"]:Destroy() end
        end

        -- 3. Aim No Cam Logic
        if _G.PvpAimNoCamActive then
            local targetPlayer = getClosestPlayer()
            if targetPlayer and targetPlayer.Character and hrp then
                local targetPart = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                if targetPart then
                    local targetPositionNoY = Vector3.new(targetPart.Position.X, hrp.Position.Y, targetPart.Position.Z)
                    hrp.CFrame = CFrame.new(hrp.Position, targetPositionNoY)
                end
            end
        end

        -- 4. Hitbox Logic
          for _, p in pairs(Players:GetPlayers()) do
        if p ~= player then
            local pChar = p.Character
            local pHrp = pChar and pChar:FindFirstChild("HumanoidRootPart")
            local pHead = pChar and pChar:FindFirstChild("Head")
            local pHum = pChar and pChar:FindFirstChildOfClass("Humanoid")

            -- Quản lý Hitbox từng người
            if _G.PvpHitboxActive and pHrp and pHum and pHum.Health > 0 then
                if pHrp.Size ~= Vector3.new(5, 5, 5) then
                    pHrp.Size = Vector3.new(5, 5, 5)
                    pHrp.CanCollide = false
                end
                if not activeHitboxes[p.Name] then
                    local box = Instance.new("BoxHandleAdornment")
                    box.Size = Vector3.new(5, 6, 5)
                    box.Color3 = Color3.fromRGB(255, 50, 50)
                    box.Transparency = 0.55
                    box.AlwaysOnTop = true
                    box.Adornee = pHrp
                    box.Parent = ScreenGui
                    activeHitboxes[p.Name] = box
                end
            else
                if activeHitboxes[p.Name] then
                    activeHitboxes[p.Name]:Destroy()
                    activeHitboxes[p.Name] = nil
                end
            end

            -- Quản lý ESP từng người
            if _G.PvpHealthActive and pHead and pHum and pHum.Health > 0 then
                if not activeHealthGuis[p.Name] then
                    createHealthGui(p, pHead)
                end
                updateHealthGui(activeHealthGuis[p.Name], pHum)
            else
                if activeHealthGuis[p.Name] then
                    activeHealthGuis[p.Name]:Destroy()
                    activeHealthGuis[p.Name] = nil
                end
            end
        end
    end
end)

local function getAllConfigFiles()
    local list = {}
    if listfiles then
        local files = listfiles(ConfigFolderName)
        for _, fullPath in ipairs(files) do
            local name = fullPath:match("([^/\\]+)%.json$") or fullPath:match("([^/\\]+)$")
            if name then table.insert(list, name) end
        end
    end
    if #list == 0 then table.insert(list, "Chưa có config nào") end
    return list
end

end

local function saveNewConfigWithName(customName)
    if customName == "" or customName == "Nhập tên cấu hình cần tạo..." then
        Fluent:Notify({Title = "Lỗi", Content = "Vui lòng nhập tên cấu hình hợp lệ trước!", Duration = 3})
        return
    end
    
    local configData = {}
    for uiName, globalVar in pairs(RegisteredToggles) do
         configData[globalVar] = _G[globalVar] or false
    end
    
    local filePath = ConfigFolderName .. "/" .. customName .. ".json"
    writefile(filePath, HttpService:JSONEncode(configData))
    
    _G.SelectedConfigName = customName
    saveGlobalSystemSettings()
    
    if Fluent.Options and Fluent.Options["SelectConfigDropdown"] then
        local newList = getAllConfigFiles()
        Fluent.Options["SelectConfigDropdown"]:SetValues(newList)
        Fluent.Options["SelectConfigDropdown"]:SetValue(customName)
    end
    
    Fluent:Notify({Title = "Hệ thống Config", Content = "Đã lưu cấu hình đơn: " .. customName, Duration = 3})
end

local function loadSelectedConfig(customName)
    if customName == "Chưa có config nào" or customName == "" then return end
    local filePath = ConfigFolderName .. "/" .. customName .. ".json"
    if isfile(filePath) then
        local fileContent = readfile(filePath)
        local success, data = pcall(function() return HttpService:JSONDecode(fileContent) end)
        
        if success and data then
             for uiName, globalVar in pairs(RegisteredToggles) do
                if data[globalVar] ~= nil then
                    _G[globalVar] = data[globalVar]
                    
                    if Fluent.Options and Fluent.Options[uiName] then
                         Fluent.Options[uiName]:SetValue(data[globalVar])
                    end
                    
                    if globalVar == "FPSBoostState" and data[globalVar] == true then activateFPSBoost() end
                    if globalVar == "Reduce20State" then toggleGraphicReduction(20, data[globalVar]) end
                    if globalVar == "Reduce40State" then toggleGraphicReduction(40, data[globalVar]) end
                    if globalVar == "Reduce60State" then toggleGraphicReduction(60, data[globalVar]) end
                    if globalVar == "Reduce100State" then toggleGraphicReduction(100, data[globalVar]) end
                    if globalVar == "ClearEverythingState" and data[globalVar] == true then
                        activateFPSBoost()
                        for _, v in ipairs(workspace:GetDescendants()) do
                            if v:IsA("Decal") or v:IsA("Texture") or v:IsA("Sound") then pcall(function() v:Destroy() end) end
                            if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") then v.Enabled = false end
                            if v:IsA("BasePart") then v.Material = Enum.Material.SmoothPlastic v.Reflectance = 0 v.CastShadow = false end
                        end
                    end
                end
            end
        end
    end
end

local function deleteSelectedConfig(customName)
    if customName == "Chưa có config nào" or customName == "" then return end
    local filePath = ConfigFolderName .. "/" .. customName .. ".json"
    if isfile(filePath) then
        delfile(filePath)
        _G.SelectedConfigName = ""
        saveGlobalSystemSettings()
        
        local newList = getAllConfigFiles()
        if Fluent.Options and Fluent.Options["SelectConfigDropdown"] then
            Fluent.Options["SelectConfigDropdown"]:SetValues(newList)
            Fluent.Options["SelectConfigDropdown"]:SetValue(newList[1] or "Chưa có config nào")
        end
        
        Fluent:Notify({Title = "Hệ thống Config", Content = "Đã xóa file cấu hình đơn: " .. customName, Duration = 3})
    else
        Fluent:Notify({Title = "Lỗi", Content = "Không tìm thấy file để xóa!", Duration = 3})
    end
end

task.defer(function()
    while not Fluent.Options or not Fluent.Options["UseConfigSystemToggle"] or not Fluent.Options["SelectConfigDropdown"] do
        task.wait(0.2)
    end
    
    if isfile(GlobalSettingsFile) then
        pcall(function()
            local data = HttpService:JSONDecode(readfile(GlobalSettingsFile))
            if data and data["SelectedConfigName"] then
                _G.SelectedConfigName = data["SelectedConfigName"]
                _G.UseConfigSystem = data["UseConfigSystem"] or false
                
                Fluent.Options["UseConfigSystemToggle"]:SetValue(_G.UseConfigSystem)
                Fluent.Options["SelectConfigDropdown"]:SetValue(_G.SelectedConfigName)
                
                if _G.UseConfigSystem then
                    loadSelectedConfig(_G.SelectedConfigName)
                    Fluent:Notify({Title = "VND💵 System", Content = "Đã tự động nạp & kích hoạt: " .. _G.SelectedConfigName, Duration = 5})
                end
            end
        end)
    end
end)
