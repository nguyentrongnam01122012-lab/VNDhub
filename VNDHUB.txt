local function isAliveModel(model)
    if model and model:FindFirstChild("Humanoid") and model.Humanoid.Health > 0 then
        return true
    end
    return false
end

local function getNearestMonster()
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local hrpPos = char.HumanoidRootPart.Position
    local closestMonster = nil
    local maxDistance = math.huge
    local targetFolder = workspace:FindFirstChild("Bosses") or workspace:FindFirstChild("Monsters") or workspace
    
    for _, obj in ipairs(targetFolder:GetChildren()) do
        if obj:IsA("Model") and isAliveModel(obj) and obj ~= char then
            if not Players:GetPlayerFromCharacter(obj) then 
                local mHrp = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Torso")
                if mHrp then
                    local distance = (hrpPos - mHrp.Position).Magnitude
                    if distance < maxDistance then
                        maxDistance = distance
                        closestMonster = obj
                    end
                end
            end
        end
    end
    return closestMonster
end

local function isAreaOccupied(targetPos)
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            if (p.Character.HumanoidRootPart.Position - targetPos).Magnitude <= 7 then
                return true
            end
        end
    end
    for _, v in ipairs(workspace:GetDescendants()) do
        if v:IsA("Seat") or v:IsA("VehicleSeat") then
            if (v.Position - targetPos).Magnitude <= 7 and v.Occupant and v.Occupant.Parent ~= player.Character then 
                return true 
            end
        end
    end
    return false
end

local function interactWithSeatPrompt(targetPos)
    for _, prompt in ipairs(workspace:GetDescendants()) do
        if prompt:IsA("ProximityPrompt") and (prompt.Parent:IsA("BasePart") or prompt.Parent:IsA("Model")) then
            if (prompt.Parent.Position - targetPos).Magnitude <= 6 then
                 fireproximityprompt(prompt)
                break
            end
        end
    end
end

local originalMaterials = {}

local function saveOriginalMaterials()
    if #originalMaterials == 0 then
        for _, v in ipairs(workspace:GetDescendants()) do
            if v:IsA("BasePart") then
                originalMaterials[v] = {
                    Material = v.Material,
                    Reflectance = v.Reflectance,
                    CastShadow = v.CastShadow
                }
            end
        end
    end
end

local function toggleGraphicReduction(percent, enable)
    saveOriginalMaterials()
    local count = 0
    for part, data in pairs(originalMaterials) do
        if part and part.Parent then
            count = count + 1
            if count % 100 <= percent then
                if enable then
                    part.Material = Enum.Material.SmoothPlastic
                    part.Reflectance = 0
                    part.CastShadow = false
                else
                    part.Material = data.Material
                    part.Reflectance = data.Reflectance
                    part.CastShadow = data.CastShadow
                end
            end
        end
    end
end

            local moveVector = Vector3.new()
            
            if UserInputService:IsKeyDown(Enum.KeyCode.W) or UserInputService:IsKeyDown(Enum.KeyCode.Up) then moveVector = moveVector + lookVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.S) or UserInputService:IsKeyDown(Enum.KeyCode.Down) then moveVector = moveVector - lookVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.A) or UserInputService:IsKeyDown(Enum.KeyCode.Left) then moveVector = moveVector - rightVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.D) or UserInputService:IsKeyDown(Enum.KeyCode.Right) then moveVector = moveVector + rightVector end
            
            if moveVector.Magnitude > 0 then freecamPosition = freecamPosition + (moveVector.Unit * flySpeed) end
            camera.CFrame = CFrame.new(freecamPosition) * camera.CFrame.Rotation
        else
            if camera.CameraType == Enum.CameraType.Scriptable then camera.CameraType = Enum.CameraType.Custom end
        end
    end)

local customSpeedMultiplier = 95  
local customJumpForce = 85        
local attackDelay = 0.1 
local activeHitboxes = {}
local activeHealthGuis = {}

local function clearAllHitboxes()
    for name, adornment in pairs(activeHitboxes) do
        if adornment then pcall(function() adornment:Destroy() end) end
    end
    activeHitboxes = {}
end

local function clearAllHealthGuis()
    for name, gui in pairs(activeHealthGuis) do
        if gui then pcall(function() gui:Destroy() end) end
    end
    activeHealthGuis = {}
end

local function getClosestPlayer()
    local closestPlayer = nil
    local shortestDistance = math.huge
    local myRoot = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    if myRoot then
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health > 0 then
                local distance = (myRoot.Position - p.Character.HumanoidRootPart.Position).Magnitude
                if distance < shortestDistance then
                    shortestDistance = distance
                    closestPlayer = p
                end
            end
        end
    end
    return closestPlayer
end

UserInputService.JumpRequest:Connect(function()
    if _G.PvpJumpActive then
        pcall(function()
            local char = player.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            if hrp and hum and hum:GetState() ~= Enum.HumanoidStateType.Freefall then
                hrp.Velocity = Vector3.new(hrp.Velocity.X, customJumpForce, hrp.Velocity.Z)
            end
        end)
    end
end)

RunService.PostSimulation:Connect(function(deltaTime)
    pcall(function()
        local char = player.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChildOfClass("Humanoid")

        if _G.PvpSpeedActive and hrp and hum and hum.MoveDirection.Magnitude > 0 then
            local velocityAddition = hum.MoveDirection * customSpeedMultiplier
            hrp.Velocity = Vector3.new(velocityAddition.X, hrp.Velocity.Y, velocityAddition.Z)
        end

        if _G.PvpSpinActive and hrp then
            local spin = hrp:FindFirstChild("VND_SpinVelocity")
            if not spin then
                spin = Instance.new("BodyAngularVelocity")
                spin.Name = "VND_SpinVelocity"
                spin.MaxTorque = Vector3.new(0, math.huge, 0)
                spin.AngularVelocity = Vector3.new(0, 35, 0) 
                spin.Parent = hrp
            end
        else
            if hrp and hrp:FindFirstChild("VND_SpinVelocity") then hrp["VND_SpinVelocity"]:Destroy() end
        end
        
        if _G.PvpAimActive then
            local targetPlayer = getClosestPlayer()
            if targetPlayer and targetPlayer.Character then
                local targetPart = targetPlayer.Character:FindFirstChild("HumanoidRootPart") or targetPlayer.Character:FindFirstChild("Head")
                if targetPart then camera.CFrame = CFrame.lookAt(camera.CFrame.Position, targetPart.Position) end
            end
        end
        
        if _G.PvpHitboxActive then
            for _, p in pairs(Players:GetPlayers()) do
                if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health > 0 then
                    local root = p.Character.HumanoidRootPart
                    if not activeHitboxes[p.Name] or not activeHitboxes[p.Name].Parent then
                        if activeHitboxes[p.Name] then activeHitboxes[p.Name]:Destroy() end
                        local box = Instance.new("BoxHandleAdornment")
                        box.Size = Vector3.new(4.5, 5.5, 4.5)
                        box.Color3 = Color3.fromRGB(255, 0, 0)
                        box.Transparency = 0.5
                        box.AlwaysOnTop = true
                        box.ZIndex = 10
                        box.Adornee = root
                        box.Parent = game.CoreGui:FindFirstChildOfClass("ScreenGui") or ScreenGui
                        activeHitboxes[p.Name] = box
                    end
                else
                    if activeHitboxes[p.Name] then activeHitboxes[p.Name]:Destroy() activeHitboxes[p.Name] = nil end
                end
            end
        else
            if next(activeHitboxes) ~= nil then clearAllHitboxes() end
        end

        if _G.PvpHealthActive then
            for _, p in pairs(Players:GetPlayers()) do
                if p ~= player and p.Character and p.Character:FindFirstChild("Head") and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health > 0 then
                    local enemyHead = p.Character.Head
                    local enemyHum = p.Character.Humanoid
                    local healthString = "❤️ " .. math.floor(enemyHum.Health) .. " / " .. math.floor(enemyHum.MaxHealth)
                    
                    local currentGui = activeHealthGuis[p.Name]
                    if not currentGui or not currentGui.Parent then
                        if currentGui then pcall(function() currentGui:Destroy() end) end
                        local bGui = Instance.new("BillboardGui")
                        bGui.Name = "VND_HealthGui"
                        bGui.Size = UDim2.new(0, 200, 0, 50)
                        bGui.StudsOffset = Vector3.new(0, 3.5, 0)
                        bGui.AlwaysOnTop = true
                        
                        local textLabel = Instance.new("TextLabel")
                        textLabel.Parent = bGui
                        textLabel.BackgroundTransparency = 1
                        textLabel.Size = UDim2.new(1, 0, 1, 0)
                        textLabel.Font = Enum.Font.GothamBold
                        textLabel.TextSize = 14
                        textLabel.TextColor3 = Color3.fromRGB(0, 255, 127)
                        textLabel.TextStrokeTransparency = 0
                        textLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
                        textLabel.Text = healthString
                        bGui.Adornee = enemyHead
                        bGui.Parent = game.CoreGui:FindFirstChildOfClass("ScreenGui") or ScreenGui
                        activeHealthGuis[p.Name] = bGui
                    else
                        local label = currentGui:FindFirstChildOfClass("TextLabel")
                        if label then
                            label.Text = healthString
                            local healthPercent = enemyHum.Health / enemyHum.MaxHealth
                            if healthPercent < 0.35 then label.TextColor3 = Color3.fromRGB(255, 30, 30)
                            elseif healthPercent < 0.7 then label.TextColor3 = Color3.fromRGB(255, 180, 0)
                            else label.TextColor3 = Color3.fromRGB(0, 255, 127) end
                        end
                    end
                else
                    if activeHealthGuis[p.Name] then activeHealthGuis[p.Name]:Destroy() activeHealthGuis[p.Name] = nil end
                end
            end
        else
            if next(activeHealthGuis) ~= nil then clearAllHealthGuis() end
        end
    end)
end)

        for _, fullPath in ipairs(files) do
            local name = fullPath:match("([^/\\]+)%.json$") or fullPath:match("([^/\\]+)$")
            if name then table.insert(list, name) end
        end
    end
    if #list == 0 then table.insert(list, "Chưa có config nào") end
    return list
end

local function saveGlobalSystemSettings()
    local globalData = {
        UseConfigSystem = _G.UseConfigSystem,
        SelectedConfigName = _G.SelectedConfigName
    }
    writefile(GlobalSettingsFile, HttpService:JSONEncode(globalData))
end

local function saveNewConfigWithName(customName)
    if customName == "" or customName == "Nhập tên cấu hình cần tạo..." or customName == "TreoMay1" then
        Fluent:Notify({Title = "Lỗi", Content = "Vui lòng nhập tên cấu hình hợp lệ trước!", Duration = 3})
        return
    end
    
    local configData = {}
    for uiName, globalVar in pairs(RegisteredToggles) do
         configData[globalVar] = _G[globalVar] or false
    end
    
    local filePath = ConfigFolderName .. "/" .. customName .. ".json"
    writefile(filePath, HttpService:JSONEncode(configData))
    
    _G.SelectedConfigName = customName
    saveGlobalSystemSettings()
    
    if Fluent.Options and Fluent.Options["SelectConfigDropdown"] then
        local newList = getAllConfigFiles()
        Fluent.Options["SelectConfigDropdown"]:SetValues(newList)
        Fluent.Options["SelectConfigDropdown"]:SetValue(customName)
    end
    
    Fluent:Notify({Title = "Hệ thống Config", Content = "Đã lưu cấu hình đơn: " .. customName, Duration = 3})
end

local function loadSelectedConfig(customName)
    if customName == "Chưa có config nào" or customName == "" then return end
    local filePath = ConfigFolderName .. "/" .. customName .. ".json"
    if isfile(filePath) then
        local fileContent = readfile(filePath)
        local success, data = pcall(function() return HttpService:JSONDecode(fileContent) end)
        
        if success and data then
             for uiName, globalVar in pairs(RegisteredToggles) do
                if data[globalVar] ~= nil then
                    _G[globalVar] = data[globalVar]
                    
                    -- 🛠️ ĐỒNG BỘ ĐỒNG THỜI: Gạt nút ảo trên UI Fluent giao diện
                    if Fluent.Options and Fluent.Options[uiName] then
                         Fluent.Options[uiName]:SetValue(data[globalVar])
                    end
                    
                    -- ⚙️ ĐỒNG BỘ LUỒNG NGẦM: Kích hoạt trực tiếp hàm hệ thống đặc biệt nếu giá trị là TRUE
                    if globalVar == "FPSBoostState" and data[globalVar] == true then activateFPSBoost() end
                    if globalVar == "Reduce20State" then toggleGraphicReduction(20, data[globalVar]) end
                    if globalVar == "Reduce40State" then toggleGraphicReduction(40, data[globalVar]) end
                    if globalVar == "Reduce60State" then toggleGraphicReduction(60, data[globalVar]) end
                    if globalVar == "Reduce100State" then toggleGraphicReduction(100, data[globalVar]) end
                    if globalVar == "ClearEverythingState" and data[globalVar] == true then
                        activateFPSBoost()
                        for _, v in ipairs(workspace:GetDescendants()) do
                            if v:IsA("Decal") or v:IsA("Texture") or v:IsA("Sound") then pcall(function() v:Destroy() end) end
                            if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") then v.Enabled = false end
                            if v:IsA("BasePart") then v.Material = Enum.Material.SmoothPlastic v.Reflectance = 0 v.CastShadow = false end
                        end
                    end
                end
            end
        end
    end
end

local function deleteSelectedConfig(customName)
    if customName == "Chưa có config nào" or customName == "" then return end
    local filePath = ConfigFolderName .. "/" .. customName .. ".json"
    if isfile(filePath) then
        delfile(filePath)
        _G.SelectedConfigName = ""
        saveGlobalSystemSettings()
        
        local newList = getAllConfigFiles()
        if Fluent.Options and Fluent.Options["SelectConfigDropdown"] then
            Fluent.Options["SelectConfigDropdown"]:SetValues(newList)
            Fluent.Options["SelectConfigDropdown"]:SetValue(newList[1] or "Chưa có config nào")
        end
        
        Fluent:Notify({Title = "Hệ thống Config", Content = "Đã xóa file cấu hình đơn: " .. customName, Duration = 3})
    else
        Fluent:Notify({Title = "Lỗi", Content = "Không tìm thấy file để xóa!", Duration = 3})
    end
end

task.spawn(function()
    -- 1. Đợi file hệ thống tồn tại
    while not isfile(GlobalSettingsFile) do task.wait(0.5) end
    
    -- 2. Đọc dữ liệu ra biến trước khi vẽ UI
    local success, data = pcall(function() return HttpService:JSONDecode(readfile(GlobalSettingsFile)) end)
    
    if success and data then
        _G.UseConfigSystem = data["UseConfigSystem"] or false
        _G.SelectedConfigName = data["SelectedConfigName"] or ""
        
        -- 3. NẾU CÓ CONFIG ĐANG CHỌN THÌ NẠP DỮ LIỆU NGAY VÀO BIẾN _G
        if _G.UseConfigSystem and _G.SelectedConfigName ~= "" and _G.SelectedConfigName ~= "Chưa có config nào" then
            local path = ConfigFolderName .. "/" .. _G.SelectedConfigName .. ".json"
            if isfile(path) then
                local configData = HttpService:JSONDecode(readfile(path))
                for key, value in pairs(configData) do
                    _G[key] = value -- Gán thẳng giá trị vào biến của game
                end
            end
        end
    end
    
    -- 4. ĐỢI UI XUẤT HIỆN RỒI MỚI GẠT NÚT (Chỉ để hiển thị cho đẹp)
    task.spawn(function()
        while not Fluent.Options or not Fluent.Options["UseConfigSystemToggle"] do task.wait(0.2) end
        if _G.UseConfigSystem then
            Fluent.Options["UseConfigSystemToggle"]:SetValue(true)
            Fluent.Options["SelectConfigDropdown"]:SetValue(_G.SelectedConfigName)
            Fluent:Notify({Title = "VND💵 System", Content = "Đã tự động kích hoạt Config: " .. _G.SelectedConfigName, Duration = 3})
        end
    end)
end)
