local function isAliveModel(model)
    if model and model:FindFirstChild("Humanoid") and model.Humanoid.Health > 0 then
        return true
    end
    return false
end

local function getNearestMonster()
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local hrpPos = char.HumanoidRootPart.Position
    local closestMonster = nil
    local maxDistance = math.huge
    local targetFolder = workspace:FindFirstChild("Bosses") or workspace:FindFirstChild("Monsters") or workspace
    
    for _, obj in ipairs(targetFolder:GetChildren()) do
        if obj:IsA("Model") and isAliveModel(obj) and obj ~= char then
            if not Players:GetPlayerFromCharacter(obj) then 
                local mHrp = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Torso")
                if mHrp then
                    local distance = (hrpPos - mHrp.Position).Magnitude
                    if distance < maxDistance then
                        maxDistance = distance
                        closestMonster = obj
                    end
                end
            end
        end
    end
    return closestMonster
end

local function isAreaOccupied(targetPos)
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            if (p.Character.HumanoidRootPart.Position - targetPos).Magnitude <= 7 then
                return true
            end
        end
    end
    for _, v in ipairs(workspace:GetDescendants()) do
        if v:IsA("Seat") or v:IsA("VehicleSeat") then
            if (v.Position - targetPos).Magnitude <= 7 and v.Occupant and v.Occupant.Parent ~= player.Character then 
                return true 
            end
        end
    end
    return false
end

local function interactWithSeatPrompt(targetPos)
    for _, prompt in ipairs(workspace:GetDescendants()) do
        if prompt:IsA("ProximityPrompt") and (prompt.Parent:IsA("BasePart") or prompt.Parent:IsA("Model")) then
            if (prompt.Parent.Position - targetPos).Magnitude <= 6 then
                 fireproximityprompt(prompt)
                break
            end
        end
    end
end

local originalMaterials = {}

local function saveOriginalMaterials()
    if #originalMaterials == 0 then
        for _, v in ipairs(workspace:GetDescendants()) do
            if v:IsA("BasePart") then
                originalMaterials[v] = {
                    Material = v.Material,
                    Reflectance = v.Reflectance,
                    CastShadow = v.CastShadow
                }
            end
        end
    end
end

local function toggleGraphicReduction(percent, enable)
    saveOriginalMaterials()
    local count = 0
    for part, data in pairs(originalMaterials) do
        if part and part.Parent then
            count = count + 1
            if count % 100 <= percent then
                if enable then
                    part.Material = Enum.Material.SmoothPlastic
                    part.Reflectance = 0
                    part.CastShadow = false
                else
                    part.Material = data.Material
                    part.Reflectance = data.Reflectance
                    part.CastShadow = data.CastShadow
                end
            end
        end
    end
end

local angle = 0
task.spawn(function()
    while true do
        task.wait(0.01)
        if _G.AutoFarmBossCircle then
            pcall(function()
                 local char = player.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                
                if hrp then
                    angle = angle + math.rad(_G.BossCircleSpeed or 5)
                     local offsetX = math.cos(angle) * (_G.BossCircleDistance or 15)
                    local offsetZ = math.sin(angle) * (_G.BossCircleDistance or 15)
                    local targetPos = Vector3.new(TDBossTam.X + offsetX, TDBossTam.Y, TDBossTam.Z + offsetZ)
                    
                    if _G.AutoAimBoss then
                        local target = getNearestMonster()
                        if target and isAliveModel(target) and target:FindFirstChild("HumanoidRootPart") then
                             local lookAtFlat = Vector3.new(target.HumanoidRootPart.Position.X, targetPos.Y, target.HumanoidRootPart.Position.Z)
                            hrp.CFrame = CFrame.lookAt(targetPos, lookAtFlat)
                        else
                            camera.CFrame = CFrame.new(targetPos)
                         end
                    else
                        hrp.CFrame = CFrame.new(targetPos)
                    end
                    
                    VirtualUser:ClickButton1(Vector2.new())
                end
            end)
        end
    end
end)

-- 🎥 Vòng lặp 8: Quản lý góc nhìn tự do
RunService.RenderStepped:Connect(function()
    pcall(function()
         if _G.LockCameraEnabled then
            camera.CameraType = Enum.CameraType.Scriptable
        elseif _G.FreecamEnabled then
            camera.CameraType = Enum.CameraType.Scriptable
            
            local lookVector = camera.CFrame.LookVector
             local rightVector = camera.CFrame.RightVector
            local moveVector = Vector3.new()
            
            if UserInputService:IsKeyDown(Enum.KeyCode.W) or UserInputService:IsKeyDown(Enum.KeyCode.Up) then moveVector = moveVector + lookVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.S) or UserInputService:IsKeyDown(Enum.KeyCode.Down) then moveVector = moveVector - lookVector end
             if UserInputService:IsKeyDown(Enum.KeyCode.A) or UserInputService:IsKeyDown(Enum.KeyCode.Left) then moveVector = moveVector - rightVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.D) or UserInputService:IsKeyDown(Enum.KeyCode.Right) then moveVector = moveVector + rightVector end
            
            if moveVector.Magnitude > 0 then
                freecamPosition = freecamPosition + (moveVector.Unit * flySpeed)
            end
             camera.CFrame = CFrame.new(freecamPosition) * camera.CFrame.Rotation
        else
            if camera.CameraType == Enum.CameraType.Scriptable then
                camera.CameraType = Enum.CameraType.Custom
            end
        end
    end)
end)
