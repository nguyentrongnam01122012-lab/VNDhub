local dragging, dragInput, dragStart, startPos
local function update(input)
	local delta = input.Position - dragStart
	ImageButton.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end
ImageButton.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = ImageButton.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then dragging = false end
		end)
	end
end)
ImageButton.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
end)
UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then update(input) end
end)

ImageButton.MouseButton1Click:Connect(function()
    Fluent:Toggle()
end)

local Players = game:GetService("Players")
local VirtualUser = game:GetService("VirtualUser")

local player = Players.LocalPlayer
local camera = workspace.CurrentCamera

_G.AutoFarmLevel = false 

local function isAliveModel(model)
    if model and model:FindFirstChild("Humanoid") and model.Humanoid.Health > 0 then
        return true
    end
    return false
end

local function getNearestMonster()
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local hrpPos = char.HumanoidRootPart.Position
    local closestMonster = nil
    local maxDistance = math.huge
    local targetFolder = workspace:FindFirstChild("Bosses") or workspace:FindFirstChild("Monsters") or workspace
    
    for _, obj in ipairs(targetFolder:GetChildren()) do
        if obj:IsA("Model") and isAliveModel(obj) and obj ~= char then
            if not Players:GetPlayerFromCharacter(obj) then 
                local mHrp = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Torso")
                if mHrp then
                    local distance = (hrpPos - mHrp.Position).Magnitude
                    if distance < maxDistance then
                        maxDistance = distance
                        closestMonster = obj
                    end
                end
            end
        end
    end
    return closestMonster
end

task.spawn(function()
    while true do
        task.wait(0.1)
        if _G.AutoFarmLevel then
            pcall(function()
                local char = player.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                local target = getNearestMonster()
                
                if hrp and target and isAliveModel(target) then
                    local targetHrp = target:FindFirstChild("HumanoidRootPart") or target:FindFirstChild("Torso")
                    if targetHrp then
                        hrp.CFrame = CFrame.new(targetHrp.Position + Vector3.new(0, 3, 0), targetHrp.Position)
                        VirtualUser:ClickButton1(Vector2.new())
                    end
                end
            end)
        end
    end
end)

local function isAreaOccupied(targetPos)
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            if (p.Character.HumanoidRootPart.Position - targetPos).Magnitude <= 7 then
                return true
            end
        end
    end
    for _, v in ipairs(workspace:GetDescendants()) do
        if v:IsA("Seat") or v:IsA("VehicleSeat") then
            if (v.Position - targetPos).Magnitude <= 7 and v.Occupant and v.Occupant.Parent ~= player.Character then 
                return true 
            end
        end
    end
    return false
end

local function interactWithSeatPrompt(targetPos)
    for _, prompt in ipairs(workspace:GetDescendants()) do
        if prompt:IsA("ProximityPrompt") and (prompt.Parent:IsA("BasePart") or prompt.Parent:IsA("Model")) then
            if (prompt.Parent.Position - targetPos).Magnitude <= 6 then
                fireproximityprompt(prompt)
                break
            end
        end
    end
end

local function activateFPSBoost()
    settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
    game:GetService("Lighting").GlobalShadows = false
    game:GetService("Lighting").Brightness = 0
    game:GetService("Lighting").FogEnd = 1e9
    for _,v in ipairs(workspace:GetDescendants()) do
        if v:IsA("Decal") or v:IsA("Texture") or v:IsA("Sound") then pcall(function() v:Destroy() end) end
        if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") then v.Enabled = false end
        if v:IsA("BasePart") then
            v.Material = Enum.Material.SmoothPlastic
            v.Reflectance = 0
            v.CastShadow = false
        end
    end
end

local function deactivateFPSBoost()
    game:GetService("Lighting").GlobalShadows = true
    game:GetService("Lighting").Brightness = 2
    game:GetService("Lighting").FogEnd = 1000
end

-- Vòng lặp 4: Farm Cao Su
local currentCaoSuTarget = nil
task.spawn(function()
    while true do
        task.wait(0.8)
        if _G.AutoFarmCaoSu then
            pcall(function()
                local char = player.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                
                if hrp then
                    if not currentCaoSuTarget or (isAreaOccupied(currentCaoSuTarget)) then
                        local chosen = nil
                        for i = 1, #TDCaoSu do
                            if not isAreaOccupied(TDCaoSu[i]) then
                                chosen = TDCaoSu[i]
                                break
                            end
                        end
                        currentCaoSuTarget = chosen or TDCaoSu[1]
                    end
                    hrp.CFrame = CFrame.new(currentCaoSuTarget + Vector3.new(0, 2, 0))
                end
            end)
        else
            currentCaoSuTarget = nil
        end
    end
end)

-- Vòng lặp 7: Logic Auto Farm Boss di chuyển vòng tròn
local angle = 0
task.spawn(function()
    while true do
        task.wait(0.01)
        if _G.AutoFarmBossCircle then
            pcall(function()
                local char = player.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                
                if hrp then
                    angle = angle + math.rad(_G.BossCircleSpeed or 5)
                    local offsetX = math.cos(angle) * (_G.BossCircleDistance or 15)
                    local offsetZ = math.sin(angle) * (_G.BossCircleDistance or 15)
                    local targetPos = Vector3.new(TDBossTam.X + offsetX, TDBossTam.Y, TDBossTam.Z + offsetZ)
                    
                    if _G.AutoAimBoss then
                        local target = getNearestMonster()
                        if target and isAliveModel(target) and target:FindFirstChild("HumanoidRootPart") then
                            local lookAtFlat = Vector3.new(target.HumanoidRootPart.Position.X, targetPos.Y, target.HumanoidRootPart.Position.Z)
                            hrp.CFrame = CFrame.lookAt(targetPos, lookAtFlat)
                        else
                            hrp.CFrame = CFrame.new(targetPos)
                        end
                    else
                        hrp.CFrame = CFrame.new(targetPos)
                    end
                    VirtualUser:ClickButton1(Vector2.new())
                end
            end)
        end
    end
end)
