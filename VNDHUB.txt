-- CẤU HÌNH LOGIC AUTO FARM QUÁI THƯỜNG
------------------------------------------------
local Players = game:GetService("Players")
local VirtualUser = game:GetService("VirtualUser")
local player = Players.LocalPlayer

_G.AutoFarmLevel = false 

local function isAliveModel(model)
    if model and model:FindFirstChild("Humanoid") and model.Humanoid.Health > 0 then
        return true
    end
    return false
end

local function getNearestMonster()
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local hrpPos = char.HumanoidRootPart.Position
    local closestMonster = nil
    local maxDistance = math.huge
    local targetFolder = workspace:FindFirstChild("Bosses") or workspace:FindFirstChild("Monsters") or workspace
    
    for _, obj in ipairs(targetFolder:GetChildren()) do
        if obj:IsA("Model") and isAliveModel(obj) and obj ~= char then
            if not Players:GetPlayerFromCharacter(obj) then 
                local mHrp = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Torso")
                if mHrp then
                    local distance = (hrpPos - mHrp.Position).Magnitude
                    if distance < maxDistance then
                        maxDistance = distance
                        closestMonster = obj
                    end
                end
            end
        end
    end
    return closestMonster
end

task.spawn(function()
    while true do
        task.wait()
        if _G.AutoFarmLevel then
            pcall(function()
                local char = player.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                local target = getNearestMonster()
                
                if hrp and target and isAliveModel(target) then
                    local targetHrp = target:FindFirstChild("HumanoidRootPart") or target:FindFirstChild("Torso")
                    if targetHrp then
                        hrp.CFrame = CFrame.new(targetHrp.Position + Vector3.new(0, 3, 0), targetHrp.Position)
                        VirtualUser:ClickButton1(Vector2.new())
                    end
                end
            end)
        end
    end
end)

------------------------------------------------
-- CẤU HÌNH LOGIC AUTO FARM Ở CHÙA (TỰ TÌM GHẾ TRỐNG)
------------------------------------------------
_G.AutoFarmChua = false

-- 2 vị trí tọa độ bạn cung cấp
local ToaDo1 = Vector3.new(1561, 133, -244)
local ToaDo2 = Vector3.new(1547, 133, -245)

-- Hàm kiểm tra xem quanh tọa độ đó có cái ghế (Seat) nào đã bị ai ngồi chưa
local function isSeatOccupied(targetPos)
    for _, v in ipairs(workspace:GetDescendants()) do
        if v:IsA("Seat") or v:IsA("VehicleSeat") then
            if (v.Position - targetPos).Magnitude <= 8 then -- Tìm cái ghế trong phạm vi 8 studs của tọa độ
                if v.Occupant then -- Nếu thuộc tính Occupant có dữ liệu => đã có người ngồi
                    return true
                end
            end
        end
    end
    return false -- Ghế trống
end

-- Vòng lặp xử lý dịch chuyển và kích hoạt bấm ngồi ở Chùa
task.spawn(function()
    while true do
        task.wait(0.3)
        if _G.AutoFarmChua then
            pcall(function()
                local char = player.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                local humanoid = char and char:FindFirstChildOfClass("Humanoid")
                
                -- Chỉ xử lý khi nhân vật chưa ngồi vào bất kỳ cái ghế nào
                if hrp and humanoid and not humanoid.SeatPart then
                    -- Mặc định chọn Tọa độ 1 để đi tới
                    local finalTarget = ToaDo1
                    
                    -- Kiểm tra xem Tọa độ 1 có người ngồi trước chưa
                    if isSeatOccupied(ToaDo1) then
                        finalTarget = ToaDo2 -- Nếu bị chiếm, tự động đổi sang Tọa độ 2
                    end
                    
                    -- Dịch chuyển nhân vật đến vị trí mục tiêu đã chọn
                    hrp.CFrame = CFrame.new(finalTarget + Vector3.new(0, 2, 0))
                    
                    -- Quét ProximityPrompt (nút nhấn) quanh vị trí đứng để kích hoạt tự ngồi
                    for _, prompt in ipairs(workspace:GetDescendants()) do
                        if prompt:IsA("ProximityPrompt") and (prompt.Parent:IsA("BasePart") or prompt.Parent:IsA("Model")) then
                            if (prompt.Parent.Position - finalTarget).Magnitude <= 6 then
                                fireproximityprompt(prompt)
                                break
                            end
                        end
                    end
                end
            end)
        end
    end
end)
